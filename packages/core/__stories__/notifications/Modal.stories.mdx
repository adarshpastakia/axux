import {
  AxBox,
  AxButton,
  AxContent,
  AxModal,
  useAxModalService
} from "@axux/core";

export const ModalBody = ({ doClose }) => {
  return (
    <AxModal onClose={doClose}>
      <AxModal.Header title="Modal title" icon="mdi mdi-bell" iconBg="primary" />
      <AxContent>Any content</AxContent>
      <AxModal.Footer>
        <AxButton onClick={doClose} />
      </AxModal.Footer>
    </AxModal>
  );
};

export const Element = () => {
  const { open } = useAxModalService();

  const showModal = () => {
    open(ModalBody);
  };

  return (
    <AxBox>
      <AxButton onClick={showModal}>Show Modal</AxButton>
    </AxBox>
  );
};
