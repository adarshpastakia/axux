import { AxButton } from "@axux/core";

export const ButtonExample = () => {
  return (
    <AxButton
      badge={{ value: "Value", color: "primary", ping: true }}
      color="primary"
      hotKey="ctrl+enter"
      icon={svgPath || "mdi mdi-icon" || "image.url"}
      iconAlign="start"
      isActive={activeState}
      isDisabled={disabledState}
      isLoading={loadingState}
      nav={routeNav}
      size="md"
      tooltip={{ content: "Tooltip", color: "info", placement: "top" }}
      type="button"
      variant="outline"
      onClick={clickHandler}
      stopPropagation
    >
      Click Me
    </AxButton>
  );
};
