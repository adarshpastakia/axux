# Overlays

<Anchor storyId="sample">
  <Subheading>Sample Code</Subheading>
</Anchor>

```tsx
import {
  useOverlayService,
  AxButton,
  AxContent,
  AxModal,
  AxFlyout,
} from "@axux/core";

const Modal = (props) => (
  <AxModal {...props}>
    <AxContent>. . .</AxContent>
  </AxModal>
);

const Flyout = (props) => (
  <AxFlyout {...props}>
    <AxContent>. . .</AxContent>
  </AxFlyout>
);

const Component = () => {
  const [modal, openModal] = useOverlayService();
  const [flyout, openFlyout] = useOverlayService(Flyout);

  return (
    <Container>
      <AxButton onClick={openModal}>Open Modal</AxButton>
      <AxButton onClick={openFlyout}>Open Flyout</AxButton>

      {modal}
      {flyout}
    </Container>
  );
};
```
