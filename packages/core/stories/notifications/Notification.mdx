```tsx
import { useNotificationService, AxButton } from "@axux/core";

const Component = () => {
  const { alert, toast, message } = useNotificationService();

  return (
    <Container>
      <AxButton onClick={() => openAlert({ message: "Test alert" })}>
        Open Alert
      </AxButton>
      <AxButton
        onClick={() => openToast({ message: "Test toast", color: "success" })}
      >
        Open Toast
      </AxButton>
      <AxButton
        onClick={() => openMessage({ message: "Test toast", color: "success" })}
      >
        Open Message
      </AxButton>
    </Container>
  );
};
```
