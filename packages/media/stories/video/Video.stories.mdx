> ### Drawing boxes and hilighting areas

```tsx
import { AxVideoPlayer, VideoPlayerRef } from "@axux/media";

const VideoTemplate = () => {
  const videoRef = useRef<VideoPlayerRef>(null);

  // draw boxes [x,y,w,h]
  videoRef.current.drawBox("0,0,96,128", { stroke: "#fc0" });
  videoRef.current.drawBox([20, 50, 96, 128], { stroke: "#fc0000" });

  // hilight image area
  videoRef.current.drawBox("40,90,96,128");

  // unhilight area and redraw original boxes
  videoRef.current.unhilight();

  // clear all boxes and hilights
  videoRef.current.clear();

  return <AxVideoPlayer src="..." ref={videoRef} />;
};
```
