# Using MapViewer

<Description of={AxMap.Viewer} />

- `shift+scroll`: zoom in/out
- `shift+drag`: zoom in
- `ctrl+drag`: draw circle selection (Requires selection tool)

```tsx
import { AxMap } from "@axux/map";

<AxMap.Viewer
    sources=[
        { id:"key", title:"Title for basemap list", thumb:"png for basemap list",
            type:"vector|raster", url:"layer source" }
    ]
    defaultSource="key"
>
    {/* Zoom tools */}
    <AxMap.tools.Zoom />
    {/* History navigation */}
    <AxMap.tools.Navigation />
    {/* Basemaps list */}
    <AxMap.tools.Basemap />
    {/* Layer list */}
    <AxMap.tools.Layers />
    {/* Selection tools */}
    <AxMap.tools.Selection
      filters="selected points"
      onUpdate={handleUpdate}
    />
    {/* Custom action */}
    <AxMap.tools.Action
        icon="mdiIcon"
        tooltip=""
        position="top-start|top-end"
        index={0}
        isDisabled={false}
        onClick={handler}
     />
    {/* Comparison tool */}
     <AxMap.tools.Comparison
        onCompareStart={() =>
          Promise.resolve({ events, compareEvents })
        }
      />

    <AxMap.layers.X />
</AxMap.Viewer>

```
